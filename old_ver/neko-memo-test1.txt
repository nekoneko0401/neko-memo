<script type="module">
    import { initializeApp } from "https://www.gstatic.com/firebasejs/10.8.0/firebase-app.js";
    import { getDatabase, ref, set, onValue } from "https://www.gstatic.com/firebasejs/10.8.0/firebase-database.js";

    // ★★★ ここに自分の firebaseConfig を貼り付けてください ★★★

        apiKey: "AIzaSyAOlqtNEaxEZ5PhX6p7yPJhv-EeIJxzk1Q",
        authDomain: "neko-memo.firebaseapp.com",
        projectId: "neko-memo",
        storageBucket: "neko-memo.firebasestorage.app",
        messagingSenderId: "267727124933",
        appId: "1:267727124933:web:74c8e7a4f968b7822f73ac"
    };

    // Firebaseを起動
    const app = initializeApp(firebaseConfig);
    const db = getDatabase(app);
    const memoRef = ref(db, 'shared-memo');

    const memoArea = document.getElementById('memoArea');
    const status = document.getElementById('status');

    // クラウドからデータが届いたら表示を更新
    onValue(memoRef, (snapshot) => {
        const data = snapshot.val();
        if (data !== null && document.activeElement !== memoArea) {
            memoArea.value = data;
        }
        status.innerText = "同期済み 🐾";
    });

    // 文字を打つたびにクラウドへ保存
    memoArea.addEventListener('input', (e) => {
        const text = e.target.value;
        set(memoRef, text);
        status.innerText = "保存中...";
    });
